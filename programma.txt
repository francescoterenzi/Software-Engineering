BusRepository.java
@Path("bus")
@Produces("text/xml")
public class BusRepository {

    private List<Bus> buses;

    @GET
    @Path("buses")
    function getBuses()
        return buses;

    @GET
    @Path("buses/{id}")
    function getBus(id)
        return buses(id)

    @PUT
    @Path("buses/{id}")
    function putBus(id)
        buses.update(id)
